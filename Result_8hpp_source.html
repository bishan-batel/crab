390 template<crab::ty::mapper<T> F>
391 [[nodiscard]] CRAB_INLINE constexpr auto and_then(
392 F&& functor,
393 const SourceLocation loc = SourceLocation::current()
394 ) && {
395 using R = ty::functor_result<F, T>;
396
397 static_assert(result_type<R>, "and_then functor parameter must return a Result<T,E> type");
398
399 static_assert(
400 std::same_as<typename R::ErrType, ErrType>,
401 "Returned result must have the same error as the original monadic "
402 "context (must be a mapping from Result<T,E> -> Result<S,E> ) "
403 );
404
405 ensure_valid(loc);
406
407 if (is_err()) {
408 return R{Err{mem::move(*this).unwrap_err(loc)}};
409 }
410
411 return std::invoke(functor, mem::move(*this).unwrap(loc));
412 }
